-	script	KoE	FAKE_NPC,{

OnInit:

        disablenpc "The King KoE#KoE";

        disablenpc "Exit#KoE";

        bindatcmd "koe", strnpcinfo(0)+"::OnCommand", 99,100;

        end;

OnCommand:

        if ( compare( .@atcmd_parameters$, "on" ) ) goto L_start;

        else if ( compare( .@atcmd_parameters$, "off" ) ) goto L_end;

        else {

                dispbottom "type - '@koe on' to start the event";

                dispbottom "type - '@koe off' to end the event";

        }

        end;

OnTue2028: //Anuncio 2 Minutos antes que inicie

        announce "King of Emperium will start in 2 minutes.", bc_all;
        end;

OnTue2058: // Anuncio que terminara en 2 Min

        announce "King of Emperium Ends in 2 minutes.", bc_all;
        end;

L_start:


OnTue2030:

        if ( .start ) end;

        gvgon "koe1";

	$KoeStatus = 1;

        announce "The King of Emperium has Begun! Enter through the NPC Located in Prontera Duration Time: 30 Minutes.", bc_all;

	query_sql("DELETE FROM `char_reg_num` WHERE `key` = 'KoeParticipante'");
	query_sql("DELETE FROM `char_reg_num` WHERE `key` = 'KoePremio'");

        .start = true;

        enablenpc "The King KoE#KoE";

        disablenpc "Exit#KoE";

        $koegid = 0;

        donpcevent "::OnRevKoE";

        maprespawnguildid "koe1", $koegid, 3;

        killmonster "koe1", "KoE::OnEmpDead";

        monster "koe1",80,65, "EMPERIUM", 1288, 1, "KoE::OnEmpDead";

        end;

L_end:


OnTue2100:

        gvgoff "koe1";

	$KoeStatus = 0;

        announce "The King of Emperium has Completed you have 2 Minutes to enter and claim the prize before the NPC disappears.", bc_all;

        .start = 0;

        enablenpc "Exit#KoE";
        killmonster "koe1", "KoE::OnEmpDead";
		sleep 120000;
        //announce "The King of Emperium ha terminado! los NPCs han desaparecido", bc_all;
		disablenpc "The King KoE#KoE";
		disablenpc "Exit#KoE";
//      maprespawnguildid "koe1", $koegid, 2; // uncomment this line to kick non-owner off the map when event ends

        end;

OnEmpDead:

        $koegid = getcharid(2);

        announce "The current King of Emperium Hill is the ["+ strcharinfo(2) +"] guild. Enter through the NPC located in Prontera.", bc_all;

        donpcevent "::OnRevKoE";

        maprespawnguildid "koe1", $koegid, 2;

        killmonster "koe1", "KoE::OnEmpDead";

        sleep 500;

        if ( .start )

                monster "koe1",80,65, "EMPERIUM", 1288, 1, "KoE::OnEmpDead";

        end;

}
